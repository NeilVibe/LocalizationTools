<!-- getMatchColor from TMTab (reuse for context results) -->
```javascript
function getMatchColor(score) {
  if (score >= 1.0) return { color: 'var(--cds-support-success, #24a148)', label: 'Exact', type: 'green' };
  else if (score >= 0.92) return { color: 'var(--cds-support-success, #24a148)', label: 'High', type: 'green' };
  else if (score >= 0.75) return { color: 'var(--cds-support-warning, #c6a300)', label: 'Fuzzy', type: 'yellow' };
  else if (score >= 0.62) return { color: 'var(--cds-support-caution-minor, #ff832b)', label: 'Low Fuzzy', type: 'orange' };
  else return { color: 'var(--cds-support-error, #da1e28)', label: 'Below Threshold', type: 'red' };
}
```
