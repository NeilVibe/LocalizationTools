```python
# Line Check improvement: group-based instead of O(n^2):
def find_inconsistencies(pairs):
    groups = defaultdict(lambda: defaultdict(int))
    for src, tgt in pairs:
        groups[src][tgt] += 1
    return {src: list(tgts.keys()) for src, tgts in groups.items() if len(tgts) > 1}

# Term Check improvement: noise filter
MAX_ISSUES_PER_TERM = 6  # If a term triggers >6 issues, it's a false positive
```
