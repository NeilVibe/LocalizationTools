    Create `tests/api/helpers/api_client.py` — a typed wrapper around FastAPI TestClient that provides named methods for every subsystem. This eliminates raw URL string duplication across test files.

    ```python
    from __future__ import annotations
    from fastapi.testclient import TestClient
    from typing import Any

    class APIClient:
        def __init__(self, client: TestClient, auth_headers: dict[str, str]):
            self.client = client
            self.headers = auth_headers

        # === Auth ===
        def login(self, username: str, password: str) -> dict:
            return self.client.post("/api/v2/auth/login", json={...}).json()

        # === Projects ===
        def create_project(self, name: str, **kwargs) -> dict: ...
        def list_projects(self) -> dict: ...
        def get_project(self, project_id: int) -> dict: ...
        def delete_project(self, project_id: int) -> dict: ...

        # === Files ===
        def upload_file(self, project_id: int, file_path: str, **kwargs) -> dict: ...
        def list_files(self, project_id: int) -> dict: ...
        def get_file(self, file_id: int) -> dict: ...
        def download_file(self, file_id: int) -> bytes: ...

        # === Rows ===
        def list_rows(self, file_id: int, **kwargs) -> dict: ...
        def update_row(self, row_id: int, **kwargs) -> dict: ...

        # === TM ===
        def create_tm(self, name: str, **kwargs) -> dict: ...
        def list_tms(self) -> dict: ...
        def search_tm(self, query: str, **kwargs) -> dict: ...

        # === GameData ===
        def browse_gamedata(self, path: str = "", depth: int = 1) -> dict: ...
        def detect_columns(self, file_path: str) -> dict: ...

        # === Codex ===
        def list_codex_types(self) -> dict: ...
        def list_codex_entities(self, entity_type: str) -> dict: ...
        def search_codex(self, query: str) -> dict: ...

        # === WorldMap ===
        def get_worldmap(self) -> dict: ...

        # === AI ===
        def ai_suggestions_status(self) -> dict: ...
        def ai_suggest(self, text: str, **kwargs) -> dict: ...

        # === Search ===
        def search_explorer(self, query: str, **kwargs) -> dict: ...
        def search_semantic(self, query: str, **kwargs) -> dict: ...

        # === QA ===
        def run_qa(self, file_id: int, **kwargs) -> dict: ...
        def qa_results(self, file_id: int) -> dict: ...

        # === Admin ===
        def admin_overview(self) -> dict: ...
        def admin_health(self) -> dict: ...
    ```

    Each method calls self.client with self.headers and returns the response object (not .json()) so callers can assert on status_code first. Include type hints throughout.

    Implement ALL methods — do not leave stubs. Read the route files to get exact endpoint paths and required parameters.
