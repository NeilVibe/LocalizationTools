    2. Add `detectEntities(node)` function:
       ```javascript
       async function detectEntities(nodeData) {
         if (!nodeData?.attributes) {
           detectedEntities = [];
           return;
         }
         // Concatenate all editable attribute values for detection
         const editableTexts = (nodeData.editable_attrs || [])
           .map(attr => nodeData.attributes[attr] || '')
           .filter(t => t.length > 0);
         if (editableTexts.length === 0) {
           detectedEntities = [];
           return;
         }
         detecting = true;
         try {
           // Detect per-attribute to get correct positions within each field
           const allDetected = [];
           for (const attr of (nodeData.editable_attrs || [])) {
             const text = nodeData.attributes[attr] || '';
             if (!text) continue;
             const response = await fetch(`${API_BASE}/api/ldm/gamedata/index/detect`, {
               method: 'POST',
               headers: { ...getAuthHeaders(), 'Content-Type': 'application/json' },
               body: JSON.stringify({ text })
             });
             if (response.ok) {
               const data = await response.json();
               for (const entity of (data.entities || [])) {
                 allDetected.push({ ...entity, attribute: attr });
               }
             }
           }
           detectedEntities = allDetected;
         } catch (err) {
           logger.error('Entity detection failed', { error: err.message });
         } finally {
           detecting = false;
         }
       }
       ```

    3. Add $effect to trigger detection when node changes:
       ```javascript
       $effect(() => {
         if (node) detectEntities(node);
       });
       ```
