       ```python
       class GameDataIndexer:
           """In-memory multi-tier index builder for gamedata entities.

           Adapted from TMIndexer. Key differences:
           - In-memory only (no disk persistence, no DB tracking)
           - Two-field extraction (name+desc) from TreeNode.attributes via EDITABLE_ATTRS
           - br-tag splitting for line tiers (not \\n)
           - Aho-Corasick automaton for Tier 0
           - Global storage (not per-tm_id)
           """

           def __init__(self):
               self._engine = None  # Lazy-loaded EmbeddingEngine
               self._indexes = None  # Dict holding all index data
               self._ready = False
               self._metadata = {}

           @property
           def is_ready(self) -> bool: ...

           @property
           def indexes(self) -> Optional[Dict]: ...

           def extract_entities_from_tree(self, roots: List[TreeNode]) -> List[Dict]:
               """Walk TreeNode hierarchy, extract entity_name + entity_desc using EDITABLE_ATTRS.

               For each node: check tag in EDITABLE_ATTRS. If present:
               - entity_name = attributes[EDITABLE_ATTRS[tag][0]] (first attr)
               - entity_desc = attributes[EDITABLE_ATTRS[tag][1]] if len > 1 else ""

               Recurse into node.children.

               Returns list of dicts: {entity_name, entity_desc, node_id, tag, file_path}
               """

           def build_indexes(self, entities: List[Dict], progress_callback=None) -> Dict:
               """Build all 6 index tiers from extracted entities.

               Pipeline (same order as TMIndexer):
               1. _build_whole_lookup(entities) -> whole_lookup dict
               2. _build_line_lookup(entities) -> line_lookup dict (br-tag split)
               3. _build_whole_embeddings(entities) -> FAISS HNSW index + mapping
               4. _build_line_embeddings(entities) -> FAISS HNSW index + mapping
               5. _build_ac_automaton(entities) -> ahocorasick.Automaton

               Store all in self._indexes dict. Set self._ready = True.
               Return metadata dict.
               """

           def build_from_folder_tree(self, folder_data: FolderTreeDataResponse) -> Dict:
               """Convenience: extract entities from FolderTreeDataResponse, then build_indexes.

               Iterates folder_data.files, for each file iterates file.roots.
               Passes file.file_path to each extracted entity dict.
               """

           def _build_whole_lookup(self, entities: List[Dict]) -> Dict[str, Dict]:
               """Hashtable: normalized entity_name -> entity data.
               Adapted from TMIndexer._build_whole_lookup.
               Key = normalize_for_hash(entity_name). No variations handling needed (gamedata names are unique).
               """

           def _build_line_lookup(self, entities: List[Dict]) -> Dict[str, Dict]:
               """Hashtable: normalized desc line -> entity data.
               Adapted from TMIndexer._build_line_lookup.
               CRITICAL: Split entity_desc by '<br/>' (NOT '\\n'). Use normalize_newlines_universal first to normalize, then split by '\\n'.
               """

           def _build_whole_embeddings(self, entities: List[Dict]) -> Dict:
               """FAISS HNSW index on concatenated (entity_name + " " + entity_desc).
               Adapted from TMIndexer._build_whole_embeddings.
               Use normalize_for_embedding on concat text.
               Use FAISSManager.build_index with in_memory=True (no path).
               Store: whole_embeddings (np.array), whole_mapping (list of entity dicts), whole_index (faiss.Index).
               """

           def _build_line_embeddings(self, entities: List[Dict]) -> Dict:
               """FAISS HNSW per desc line after br-tag split.
               Same br-tag split logic as _build_line_lookup.
               """

           def _build_ac_automaton(self, entities: List[Dict]) -> "ahocorasick.Automaton":
               """Build Aho-Corasick automaton from all entity_name values.
               Pattern from GlossaryService.build_from_entity_names:
               - Create ahocorasick.Automaton()
               - For each entity: automaton.add_word(entity_name, (idx, entity_name))
               - automaton.make_automaton()
               - Use frozenset dedup (from QuickCheck pattern) to skip duplicate names.
               """

           def get_status(self) -> Dict:
               """Return index status: ready, entity_count, build_time_ms, tier counts."""

           def clear(self):
               """Reset all indexes. Called before rebuild on new folder load."""
       ```

       ```python
       class GameDataSearcher:
           """6-Tier Cascade search for gamedata entities.

           Adapted from TMSearcher. Adds Tier 0 (Aho-Corasick) and adapts
           result schema for gamedata (node_id, tag instead of entry_id).

           Tier 0: Aho-Corasick entity detection (O(n) multi-pattern match)
           Tier 1: Perfect whole match (hash) -- exact entity name
           Tier 2: Whole embedding match (FAISS HNSW) -- semantic on name+desc
           Tier 3: Perfect line match (hash) -- exact desc line after br-split
           Tier 4: Line embedding match (FAISS HNSW) -- semantic on desc lines
           Tier 5: N-gram fallback -- Jaccard trigram on entity names
           """

           DEFAULT_THRESHOLD = 0.92

           def __init__(self, indexes: Dict[str, Any], model=None, threshold: float = DEFAULT_THRESHOLD):
               """Initialize with indexes dict from GameDataIndexer.build_indexes().

               Extract: whole_lookup, line_lookup, whole_index, line_index,
               whole_mapping, line_mapping, ac_automaton from indexes dict.
               """
               self.indexes = indexes
               self.threshold = threshold
               self._engine = model

               self.whole_lookup = indexes.get("whole_lookup", {})
               self.line_lookup = indexes.get("line_lookup", {})
               self.whole_index = indexes.get("whole_index")
               self.line_index = indexes.get("line_index")
               self.whole_mapping = indexes.get("whole_mapping", [])
               self.line_mapping = indexes.get("line_mapping", [])
               self.ac_automaton = indexes.get("ac_automaton")

           def search(self, query: str, top_k: int = 5, threshold: float = None) -> Dict[str, Any]:
               """6-Tier Cascade search. Same structure as TMSearcher.search but with Tier 0.

               Returns: {tier: int, tier_name: str, results: List[Dict], perfect_match: bool}

               Each result dict: {entity_name, entity_desc, node_id, tag, file_path, score, match_type}

               Tier 1: normalize_for_hash(query) lookup in whole_lookup
               Tier 2: encode query, FAISS search whole_index, filter by threshold
               Tier 3: split query by \\n, normalize_for_hash each line, lookup in line_lookup
               Tier 4: encode each query line, FAISS search line_index
               Tier 5: _ngram_search on whole_lookup keys (same as TMSearcher._ngram_search)
               """

           def detect_entities(self, text: str) -> List[Dict]:
               """Tier 0: Aho-Corasick scan for entity names in text.

               Uses ac_automaton.iter(text) pattern from GlossaryService.detect_entities.
               Returns list of {term, start, end, node_id, tag, entity_name}.
               Uses is_isolated() from qa_helpers for word boundary check.
               """

           def _ngram_search(self, query: str, top_k: int, threshold: float, n: int = 3) -> List[Dict]:
               """Jaccard trigram search. Same logic as TMSearcher._ngram_search.
               Adapted result schema: entity_name, entity_desc, node_id, tag instead of entry_id.
               """

           def _get_ngrams(self, text: str, n: int) -> set:
               """Same as TMSearcher._get_ngrams."""
       ```
