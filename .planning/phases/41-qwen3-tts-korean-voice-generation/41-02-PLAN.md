    **4. Add generateVoice function (after generateImage function):**
    ```javascript
    async function generateVoice(force = false) {
      if (!entity) return;
      generatingVoice = true;
      voiceGenError = null;
      try {
        const response = await fetch(
          `${API_BASE}/api/ldm/codex/generate-voice/${entity.strkey}?force=${force}`,
          { method: 'POST', headers: getAuthHeaders() }
        );
        if (!response.ok) {
          const err = await response.json().catch(() => ({}));
          throw new Error(err.detail || `HTTP ${response.status}`);
        }
        const data = await response.json();
        generatedAudioUrl = `${API_BASE}${data.audio_url}?t=${Date.now()}`;
        voiceDuration = data.duration_ms;
        audioError = false;
        logger.info('Voice generated', { strkey: entity.strkey, status: data.status, duration: data.duration_ms });
      } catch (err) {
        logger.error('Voice generation failed', { error: err.message });
        voiceGenError = err.message || 'Voice generation failed';
      } finally {
        generatingVoice = false;
      }
    }
    ```
