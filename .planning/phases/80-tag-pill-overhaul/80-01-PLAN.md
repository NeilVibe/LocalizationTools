    2. Add as HIGHEST priority pattern (before staticinfo):
       ```
       {
         name: 'combinedcolor',
         // Priority 0: PAColor wrapping a code -- HTML-escaped angle brackets
         // Matches: &lt;PAColor0xHEX&gt;{code}&lt;PAOldColor&gt;
         // Also matches param inside: &lt;PAColor0xHEX&gt;%N#&lt;PAOldColor&gt;
         regex: /&lt;PAColor(0x[0-9a-fA-F]{6,8})&gt;(?:(\{[^}]+\})|(&#37;\d#))&lt;PAOldColor&gt;/g,
         // Note: Use &#37; for % if needed, or just %
         format: (m) => {
           const hexCode = m[1];
           const cssColor = hexToCSS(hexCode); // reuse from colorParser
           const innerCode = m[2] || m[3]; // braced or param
           let label;
           if (m[2]) {
             // Braced: extract content between { }
             label = m[2].slice(1, -1);
           } else {
             // Param: %N# -> ParamN
             label = 'Param' + m[3].charAt(1);
           }
           return { label, type: 'combinedcolor', cssColor };
         },
         color: 'combined'  // Special: TagText uses cssColor for dynamic coloring
       }
       ```
