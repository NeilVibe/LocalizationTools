**1. Progressive loading — split single context fetch into separate parallel fetches:**

Replace the single `/gamedata/context` fetch with parallel independent fetches:

```javascript
// Cross-refs: fast (hash lookup), fetch immediately
let crossRefsData = $state(null);
let crossRefsLoading = $state(false);

// Related: medium speed (FAISS), fetch immediately

let relatedLoading = $state(false);

// Media: fast (attribute check), fetch immediately
let mediaData = $state(null);
let mediaLoading = $state(false);

// AI summary: on-demand only
let aiSummary = $state('');
let aiLoading = $state(false);
```

In the `$effect` watching `node`, launch 3 parallel fetches:
```javascript
$effect(() => {
  if (!node) { /* reset all state */ return; }

  const cached = contextCache.get(node.node_id);
  if (cached) {
    crossRefsData = cached.cross_refs;
    relatedData = cached.related;
    mediaData = cached.media;
    return;
  }

  const controller = new AbortController();
  const body = JSON.stringify({
    node_id: node.node_id, tag: node.tag,
    attributes: node.attributes, editable_attrs: node.editable_attrs
  });
  const headers = { ...getAuthHeaders(), 'Content-Type': 'application/json' };

  // Single fetch to /gamedata/context (returns all at once)
  // BUT render progressively: show cross-refs section first, then related, then media
  crossRefsLoading = true;
  relatedLoading = true;
  mediaLoading = true;

  fetch(`${API_BASE}/api/ldm/gamedata/context`, {
    method: 'POST', headers, body, signal: controller.signal
  })
    .then(r => r.ok ? r.json() : Promise.reject(new Error(`HTTP ${r.status}`)))
    .then(data => {
      crossRefsData = data.cross_refs;
      crossRefsLoading = false;
      // Slight stagger for visual progressive effect
      setTimeout(() => { relatedData = data.related; relatedLoading = false; }, 50);
      setTimeout(() => { mediaData = data.media; mediaLoading = false; }, 100);
      // Cache
      contextCache.set(node.node_id, data);
    })
    .catch(err => {
      if (err.name !== 'AbortError') {
        contextError = err.message;
        crossRefsLoading = false; relatedLoading = false; mediaLoading = false;
      }
    });

  return () => controller.abort();
});
```

`requestAISummary` function:
```javascript
async function requestAISummary() {
  if (!node || aiLoading) return;
  aiLoading = true;
  try {
    const response = await fetch(`${API_BASE}/api/ldm/gamedata/context/ai-summary`, {
      method: 'POST',
      headers: { ...getAuthHeaders(), 'Content-Type': 'application/json' },
      body: JSON.stringify({
        node_id: node.node_id, tag: node.tag,
        attributes: node.attributes, editable_attrs: node.editable_attrs
      })
    });
    if (response.ok) {
      const data = await response.json();
      aiSummary = data.summary;
      aiAvailable = data.available;
    }
  } catch (err) {
    logger.error('AI summary failed', { error: err.message });
  } finally {
    aiLoading = false;
  }
}
```

Reset aiSummary when node changes (in the $effect).
