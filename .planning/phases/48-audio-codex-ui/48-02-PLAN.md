<!-- Backend API endpoints (from Plan 01) -->
```
GET /api/ldm/codex/audio?offset=0&limit=50&category=Dialog/QuestDialog&q=search_term
  -> { items: AudioCardResponse[], total, offset, limit, has_more, category_filter }

GET /api/ldm/codex/audio/categories
  -> { categories: AudioCategoryNode[], total_events }
  AudioCategoryNode: { name, full_path, count, children: AudioCategoryNode[] }

GET /api/ldm/codex/audio/{event_name}
  -> { event_name, string_id, script_kr, script_eng, export_path, has_wem, xml_order, wem_path, stream_url }

GET /api/ldm/codex/audio/stream/{event_name}
  -> audio/wav binary (FileResponse)
```

<!-- AudioCardResponse shape -->
```typescript
interface AudioCard {
  event_name: string;
  string_id: string | null;
  script_kr: string | null;
  script_eng: string | null;
  export_path: string | null;
  has_wem: boolean;
  xml_order: number | null;
}
```
