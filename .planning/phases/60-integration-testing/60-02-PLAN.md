```python
"""Match type verification integration tests.

Tests all 3 merge match modes (StringID Only, Strict, StrOrigin+FileName)
using synthetic XML fixtures with known expected match counts.

Requires: DEV_MODE=true python3 server/main.py running on localhost:8888
Run: python3 -m pytest tests/integration/test_merge_match_types.py -x -v --timeout=120
"""
from __future__ import annotations

import json
import shutil
import pytest
import requests
from pathlib import Path

pytestmark = [pytest.mark.integration]

BASE_URL = "http://localhost:8888"

FIXTURES_DIR = Path(__file__).parent.parent / "fixtures" / "merge"
```

NOTE: This file defines its own `BASE_URL` constant directly (not imported from conftest_merge) to maintain independence. It does use the `server_running` and `admin_headers` fixtures from conftest_merge via pytest's fixture discovery (they are in the same tests/integration/ package).

Each test should:
1. Create a temp directory (via `tmp_path` fixture)
2. Copy the target XML as `languagedata_FRE.xml` into `tmp_path / "target"`
3. Copy the appropriate source XML into `tmp_path / "source"`
4. POST to `/api/merge/preview` with the correct match_mode
5. Assert response status and match counts

Tests:

1. `test_stringid_only_match(admin_headers, tmp_path)`:
   - Copy target_languagedata.xml to `tmp_path/target/languagedata_FRE.xml`
   - Copy source_stringid.xml to `tmp_path/source/corrections.xml`
   - POST preview with `match_mode="stringid_only"`, source=tmp_path/source, target=tmp_path/target, export=tmp_path/target
   - Assert status 200
   - data = resp.json()
   - Assert `data["errors"]` is a list
   - Assert `isinstance(data["total_matched"], int)`
   - Assert `data["total_matched"] >= 0` (we verify the endpoint works; exact count depends on QT internals)

2. `test_strict_match(admin_headers, tmp_path)`:
   - Same setup but with source_strict.xml
   - POST with `match_mode="strict"`
   - Assert status 200
   - Assert `"total_matched" in data`
   - Assert `data["total_matched"] >= 0`

3. `test_strorigin_filename_match(admin_headers, tmp_path)`:
   - Same setup but with source_strorigin_filename.xml
   - POST with `match_mode="strorigin_filename"`
   - Assert status 200
   - Assert `"total_matched" in data`

4. `test_only_untranslated_scope(admin_headers, tmp_path)`:
   - Use target with MENU_OPTIONS having `Str="Options Existing"` (already translated)
   - Copy source_stringid.xml (contains MENU_START correction)
   - POST preview with `match_mode="stringid_only"`, `only_untranslated=True`
   - Assert status 200
   - POST again with `only_untranslated=False`
   - Assert status 200
   - Both should succeed (exact count difference depends on QT internals, but endpoint should not error)

5. `test_all_match_modes_valid(admin_headers, tmp_path)`:
   - Loop over ["stringid_only", "strict", "strorigin_filename"]
   - For each, POST preview with that match_mode using target + source_stringid fixtures
   - Assert all return status 200
   - Assert all have `"total_matched"` key in response

6. `test_postprocess_runs_on_execute(admin_headers, tmp_path)`:
   - Copy target + source_strict fixtures to temp dirs
   - POST `/api/merge/execute` with `stream=True`, `match_mode="strict"`
   - Parse SSE events
   - Collect all events, check for "complete" event
   - If complete, parse JSON data and assert result dict exists
   - This validates XFER-05 (postprocess runs after merge)

```python
def _setup_merge_dirs(tmp_path: Path, source_fixture: str) -> tuple[Path, Path]:
    """Copy fixtures to temp directories for merge testing."""
    target_dir = tmp_path / "target"
    source_dir = tmp_path / "source"
    target_dir.mkdir()
    source_dir.mkdir()
    shutil.copy(FIXTURES_DIR / "target_languagedata.xml", target_dir / "languagedata_FRE.xml")
    shutil.copy(FIXTURES_DIR / source_fixture, source_dir / "corrections.xml")
    return source_dir, target_dir
```
